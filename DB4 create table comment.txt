CREATE TABLE retable (
co_num VARCHAR2(10) PRIMARY KEY,
bo_num VARCHAR2(10),
name varchar2(100),
rememo varchar2(1000),
time varchar2(100)
);

CREATE SEQUENCE retable_seq START WITH 1 INCREMENT BY 1;

CREATE TRIGGER retable_trigger
BEFORE INSERT ON retable
FOR EACH ROW
BEGIN
  SELECT retable_seq.NEXTVAL
  INTO   :new.co_num
  FROM   dual;
END;
/
